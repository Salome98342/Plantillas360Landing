            const mobileBtnIcon = mobileBtn.querySelector('i');

            mobileBtn.addEventListener('click', () => {
                navLinks.classList.toggle('active');
                
                // Cambiar icono de hamburguesa a "X"
                if(navLinks.classList.contains('active')) {
                    mobileBtnIcon.classList.remove('fa-bars');
                    mobileBtnIcon.classList.add('fa-xmark');
                } else {
                    mobileBtnIcon.classList.remove('fa-xmark');
                    mobileBtnIcon.classList.add('fa-bars');
                }
            });

            // Cerrar menú móvil al hacer clic en un enlace
            const navItems = navLinks.querySelectorAll('a');
            navItems.forEach(item => {
                item.addEventListener('click', () => {
                    if (window.innerWidth <= 768) {
                        navLinks.classList.remove('active');
                        mobileBtnIcon.classList.remove('fa-xmark');
                        mobileBtnIcon.classList.add('fa-bars');
                    }
                });
            });

            // 2. Animaciones al hacer scroll (Intersection Observer)
            // Esto detecta cuando un elemento entra en la pantalla y le añade la clase 'is-visible'
            const observerOptions = {
                root: null,
                rootMargin: '0px',
                threshold: 0.15 // El 15% del elemento debe ser visible para disparar la animación
            };

            const observer = new IntersectionObserver((entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('is-visible');
                        // Dejamos de observar el elemento una vez que ya se animó para mejor rendimiento
                        observer.unobserve(entry.target); 
                    }
                });
            }, observerOptions);

            const animatedElements = document.querySelectorAll('.animate-on-scroll');
            animatedElements.forEach(el => observer.observe(el));

            // 3. Actualizar año automático en el Footer
            document.getElementById('currentYear').textContent = new Date().getFullYear();
        });
    </script>
